SharePoint CSV Agent — Python Code

#!/usr/bin/env python3
"""
SharePoint CSV Consolidation Agent

Monitors a SharePoint document library for new dated subfolders containing CSV files.
Waits 30 minutes after the first file is detected, then consolidates all CSVs into a
single normalized output file and uploads it back to SharePoint.

Usage:
    python sharepoint_csv_agent.py                     # Normal run (poll once and exit)
    python sharepoint_csv_agent.py --reprocess 2026-03-18  # Force reprocess a specific date
    python sharepoint_csv_agent.py --config /path/to/config.json

Dependencies:
    pip install msal requests pandas

Auth:
    Requires an Azure AD app registration with Sites.Read.All and Files.ReadWrite.All
    application permissions. Set credentials via config.json or environment variables.
"""

import argparse
import csv
import io
import json
import logging
import os
import re
import sys
import tempfile
import time
from datetime import datetime, timedelta, timezone
from pathlib import Path

import msal
import pandas as pd
import requests

# ---------------------------------------------------------------------------
# Configuration
# ---------------------------------------------------------------------------

DEFAULT_CONFIG = {
    "tenant_id": "",
    "client_id": "",
    "client_secret": "",
    "site_url": "",                          # e.g. https://contoso.sharepoint.com/sites/DataTeam
    "drop_folder": "Shared Documents/CSVDrops",
    "output_folder": "Shared Documents/CSVConsolidated",
    "date_format": "%Y-%m-%d",               # Expected date folder naming format
    "date_pattern": r"^\d{4}-\d{2}-\d{2}$",  # Regex to identify dated folders
    "wait_minutes": 30,
    "state_file": "agent_state.json",
    "log_file": "agent.log",
    "log_level": "INFO",
    "add_source_metadata": False,            # Add _source_file column to output
    "lock_file": "agent.lock",
    "max_retries": 3,
    "retry_backoff_base": 2,                 # Exponential backoff base in seconds
}

GRAPH_BASE = "https://graph.microsoft.com/v1.0"
GRAPH_SCOPE = ["https://graph.microsoft.com/.default"]


def load_config(config_path: str | None = None) -> dict:
    """Load configuration from file and overlay with environment variables."""
    config = dict(DEFAULT_CONFIG)

    # Load from file if provided
    if config_path and os.path.exists(config_path):
        with open(config_path, "r") as f:
            file_config = json.load(f)
        config.update(file_config)

    # Environment variables override file config
    env_map = {
        "SP_TENANT_ID": "tenant_id",
        "SP_CLIENT_ID": "client_id",
        "SP_CLIENT_SECRET": "client_secret",
        "SP_SITE_URL": "site_url",
        "SP_DROP_FOLDER": "drop_folder",
        "SP_OUTPUT_FOLDER": "output_folder",
        "SP_DATE_FORMAT": "date_format",
        "SP_WAIT_MINUTES": "wait_minutes",
        "SP_STATE_FILE": "state_file",
        "SP_LOG_FILE": "log_file",
    }
    for env_var, config_key in env_map.items():
        val = os.environ.get(env_var)
        if val is not None:
            if config_key == "wait_minutes":
                val = int(val)
            config[config_key] = val

    return config

def setup_logging(config: dict) -> logging.Logger:
    """Configure structured logging to file and stdout."""
    logger = logging.getLogger("csv_agent")
    logger.setLevel(getattr(logging, config["log_level"].upper(), logging.INFO))

    formatter = logging.Formatter(
        '{"time": "%(asctime)s", "level": "%(levelname)s", "message": "%(message)s"}',
        datefmt="%Y-%m-%dT%H:%M:%S%z",
    )

    # File handler with rotation-friendly naming
    if config.get("log_file"):
        fh = logging.FileHandler(config["log_file"])
        fh.setFormatter(formatter)
        logger.addHandler(fh)

    # Stdout handler
    sh = logging.StreamHandler(sys.stdout)
    sh.setFormatter(formatter)
    logger.addHandler(sh)

    return logger

class FileLock:
    """Simple file-based lock to prevent overlapping cron executions."""

    def __init__(self, lock_path: str):
        self.lock_path = lock_path
        self.locked = False

    def acquire(self) -> bool:
        if os.path.exists(self.lock_path):
            # Check if the lock is stale (older than 1 hour)
            try:
                mtime = os.path.getmtime(self.lock_path)
                if time.time() - mtime > 3600:
                    os.remove(self.lock_path)
                else:
                    return False
            except OSError:
                return False

        try:
            with open(self.lock_path, "w") as f:
                f.write(str(os.getpid()))
            self.locked = True
            return True
        except OSError:
            return False

    def release(self):
        if self.locked and os.path.exists(self.lock_path):
            os.remove(self.lock_path)
            self.locked = False

def load_state(state_path: str) -> dict:
    """Load agent state from JSON file. Returns empty dict if not found."""
    if os.path.exists(state_path):
        with open(state_path, "r") as f:
            return json.load(f)
    return {}


def save_state(state_path: str, state: dict):
    """Persist agent state to JSON file atomically."""
    tmp_path = state_path + ".tmp"
    with open(tmp_path, "w") as f:
        json.dump(state, f, indent=2, default=str)
    os.replace(tmp_path, state_path)

class GraphClient:
    """Thin wrapper around Microsoft Graph API for SharePoint operations."""

    def __init__(self, config: dict, logger: logging.Logger):
        self.config = config
        self.logger = logger
        self._token: str | None = None
        self._token_expiry: datetime | None = None
        self._site_id: str | None = None
        self._drive_id: str | None = None

    # -- Authentication --

    def _get_token(self) -> str:
        """Acquire or refresh an access token using MSAL client credentials flow."""
        if self._token and self._token_expiry and datetime.now(timezone.utc) < self._token_expiry:
            return self._token

        authority = f"https://login.microsoftonline.com/{self.config['tenant_id']}"
        app = msal.ConfidentialClientApplication(
            self.config["client_id"],
            authority=authority,
            client_credential=self.config["client_secret"],
        )

        result = app.acquire_token_for_client(scopes=GRAPH_SCOPE)

        if "access_token" not in result:
            error_desc = result.get("error_description", "Unknown error")
            raise RuntimeError(f"Auth failure: {error_desc}")

        self._token = result["access_token"]
        # Tokens typically last 1 hour; refresh 5 minutes early
        self._token_expiry = datetime.now(timezone.utc) + timedelta(minutes=55)
        self.logger.info("Acquired new access token")
        return self._token

    def _headers(self) -> dict:
        return {"Authorization": f"Bearer {self._get_token()}"}

    # -- API helpers with retry --

    def _request(self, method: str, url: str, **kwargs) -> requests.Response:
        """Make an HTTP request with retry and exponential backoff."""
        max_retries = self.config.get("max_retries", 3)
        backoff_base = self.config.get("retry_backoff_base", 2)

        for attempt in range(max_retries):
            try:
                resp = requests.request(method, url, headers=self._headers(), timeout=60, **kwargs)
                if resp.status_code == 429:
                    # Respect Retry-After header from throttling
                    retry_after = int(resp.headers.get("Retry-After", backoff_base ** attempt))
                    self.logger.warning(f"Throttled (429). Retrying in {retry_after}s")
                    time.sleep(retry_after)
                    continue
                resp.raise_for_status()
                return resp
            except requests.exceptions.RequestException as e:
                if attempt < max_retries - 1:
                    wait = backoff_base ** attempt
                    self.logger.warning(f"Request failed (attempt {attempt + 1}): {e}. Retrying in {wait}s")
                    time.sleep(wait)
                else:
                    raise
        # Should not reach here, but satisfy type checker
        raise RuntimeError("Max retries exceeded")

    # -- SharePoint site and drive resolution --

    def _resolve_site_id(self) -> str:
        """Resolve the SharePoint site URL to a Graph site ID."""
        if self._site_id:
            return self._site_id

        site_url = self.config["site_url"].rstrip("/")
        # Parse hostname and site path from URL
        # e.g. https://contoso.sharepoint.com/sites/DataTeam
        from urllib.parse import urlparse
        parsed = urlparse(site_url)
        hostname = parsed.hostname
        site_path = parsed.path.rstrip("/")

        url = f"{GRAPH_BASE}/sites/{hostname}:{site_path}"
        resp = self._request("GET", url)
        self._site_id = resp.json()["id"]
        self.logger.debug(f"Resolved site ID: {self._site_id}")
        return self._site_id

    def _resolve_drive_id(self) -> str:
        """Get the default document library drive ID for the site."""
        if self._drive_id:
            return self._drive_id

        site_id = self._resolve_site_id()
        url = f"{GRAPH_BASE}/sites/{site_id}/drive"
        resp = self._request("GET", url)
        self._drive_id = resp.json()["id"]
        self.logger.debug(f"Resolved drive ID: {self._drive_id}")
        return self._drive_id

    # -- Folder and file operations --

    def list_subfolders(self, folder_path: str) -> list[dict]:
        """List immediate child folders under the given path.

        Returns a list of dicts with 'name' and 'id' keys.
        """
        drive_id = self._resolve_drive_id()
        # URL-encode the path for the Graph API (colons approach)
        url = f"{GRAPH_BASE}/drives/{drive_id}/root:/{folder_path}:/children"
        params = {"$filter": "folder ne null", "$select": "name,id,folder,createdDateTime"}

        resp = self._request("GET", url, params=params)
        data = resp.json()

        folders = []
        for item in data.get("value", []):
            if "folder" in item:
                folders.append({
                    "name": item["name"],
                    "id": item["id"],
                    "created": item.get("createdDateTime", ""),
                })
        return folders

    def list_files(self, folder_path: str, extension: str = ".csv") -> list[dict]:
        """List files in a folder, optionally filtered by extension.

        Returns a list of dicts with 'name', 'id', and 'size' keys.
        """
        drive_id = self._resolve_drive_id()
        url = f"{GRAPH_BASE}/drives/{drive_id}/root:/{folder_path}:/children"
        params = {"$select": "name,id,size,file"}

        resp = self._request("GET", url, params=params)
        data = resp.json()

        files = []
        for item in data.get("value", []):
            if "file" in item and item["name"].lower().endswith(extension):
                files.append({
                    "name": item["name"],
                    "id": item["id"],
                    "size": item.get("size", 0),
                })
        return files

    def download_file(self, file_id: str) -> bytes:
        """Download a file's content by its item ID."""
        drive_id = self._resolve_drive_id()
        url = f"{GRAPH_BASE}/drives/{drive_id}/items/{file_id}/content"
        resp = self._request("GET", url)
        return resp.content

    def upload_file(self, folder_path: str, filename: str, content: bytes) -> dict:
        """Upload a file to a SharePoint folder. Overwrites if exists.

        For files > 4 MB, uses an upload session. For smaller files, uses simple PUT.
        """
        drive_id = self._resolve_drive_id()
        file_path = f"{folder_path}/{filename}"

        if len(content) > 4 * 1024 * 1024:
            return self._upload_large_file(drive_id, file_path, content)

        url = f"{GRAPH_BASE}/drives/{drive_id}/root:/{file_path}:/content"
        resp = self._request("PUT", url, data=content,
                             headers={**self._headers(), "Content-Type": "text/csv"})
        # PUT with content sets the Content-Type header; need to merge with auth
        # Actually, _request sets auth headers internally — override here
        resp_data = resp.json() if resp.content else {}
        self.logger.info(f"Uploaded {filename} ({len(content)} bytes)")
        return resp_data

    def _upload_large_file(self, drive_id: str, file_path: str, content: bytes) -> dict:
        """Upload a large file using an upload session (for files > 4 MB)."""
        url = f"{GRAPH_BASE}/drives/{drive_id}/root:/{file_path}:/createUploadSession"
        session_resp = self._request("POST", url, json={
            "item": {"@microsoft.graph.conflictBehavior": "replace"}
        })
        upload_url = session_resp.json()["uploadUrl"]

        # Upload in 10 MB chunks
        chunk_size = 10 * 1024 * 1024
        total_size = len(content)
        result = {}

        for start in range(0, total_size, chunk_size):
            end = min(start + chunk_size, total_size)
            chunk = content[start:end]
            headers = {
                "Content-Range": f"bytes {start}-{end - 1}/{total_size}",
                "Content-Length": str(len(chunk)),
            }
            resp = requests.put(upload_url, data=chunk, headers=headers, timeout=120)
            resp.raise_for_status()
            if resp.status_code in (200, 201):
                result = resp.json()

        self.logger.info(f"Uploaded large file {file_path} ({total_size} bytes)")
        return result

    def upload_file_with_auth(self, folder_path: str, filename: str, content: bytes) -> dict:
        """Upload with proper auth header merging."""
        drive_id = self._resolve_drive_id()
        file_path = f"{folder_path}/{filename}"

        if len(content) > 4 * 1024 * 1024:
            return self._upload_large_file(drive_id, file_path, content)

        url = f"{GRAPH_BASE}/drives/{drive_id}/root:/{file_path}:/content"
        headers = self._headers()
        headers["Content-Type"] = "text/csv"
        resp = requests.put(url, data=content, headers=headers, timeout=120)
        resp.raise_for_status()
        self.logger.info(f"Uploaded {filename} ({len(content)} bytes)")
        return resp.json() if resp.content else {}

def consolidate_csvs(
    csv_contents: list[tuple[str, bytes]],
    add_source_metadata: bool = False,
    logger: logging.Logger | None = None,
) -> bytes:
    """Consolidate multiple CSV file contents into a single normalized CSV.

    Args:
        csv_contents: List of (filename, raw_bytes) tuples.
        add_source_metadata: If True, add a _source_file column.
        logger: Logger instance.

    Returns:
        The consolidated CSV as UTF-8 bytes (with BOM for Excel/Tableau compatibility).
    """
    if not logger:
        logger = logging.getLogger("csv_agent")

    dataframes = []
    skipped_files = []

    for filename, raw_bytes in csv_contents:
        try:
            # Detect encoding — try UTF-8 first, fall back to latin-1
            try:
                text = raw_bytes.decode("utf-8-sig")  # Handle BOM
            except UnicodeDecodeError:
                text = raw_bytes.decode("latin-1")

            # Read CSV with all values as strings to avoid any type coercion
            df = pd.read_csv(io.StringIO(text), dtype=str, keep_default_na=False)

            if add_source_metadata:
                df["_source_file"] = filename

            dataframes.append(df)
            logger.info(f"Read {filename}: {len(df)} rows, {len(df.columns)} columns")

        except Exception as e:
            logger.warning(f"Skipped {filename}: {e}")
            skipped_files.append(filename)

    if not dataframes:
        raise ValueError("No CSV files could be parsed. All files were skipped.")

    # Concatenate all DataFrames — union of all columns, missing values become ""
    consolidated = pd.concat(dataframes, ignore_index=True, sort=False)
    consolidated = consolidated.fillna("")

    logger.info(
        f"Consolidated: {len(consolidated)} total rows, "
        f"{len(consolidated.columns)} columns, "
        f"{len(dataframes)} files processed, {len(skipped_files)} files skipped"
    )

    if skipped_files:
        logger.warning(f"Skipped files: {', '.join(skipped_files)}")

    # Write to CSV bytes with UTF-8 BOM for Tableau/Excel compatibility
    output = io.BytesIO()
    output.write(b"\xef\xbb\xbf")  # UTF-8 BOM
    consolidated.to_csv(output, index=False, quoting=csv.QUOTE_ALL, encoding="utf-8")
    return output.getvalue()

class SharePointCSVAgent:
    """Orchestrates the monitor → wait → consolidate → upload pipeline."""

    def __init__(self, config: dict, logger: logging.Logger):
        self.config = config
        self.logger = logger
        self.graph = GraphClient(config, logger)
        self.state_path = config["state_file"]
        self.state = load_state(self.state_path)

    def run(self, reprocess_date: str | None = None):
        """Execute one polling cycle of the agent.

        This method is designed to be called repeatedly (e.g., every 5 minutes via cron).
        It is idempotent and safe to call concurrently (with file locking).
        """
        self.logger.info("Agent run started")

        if reprocess_date:
            self._handle_reprocess(reprocess_date)
            return

        # Step 1: List dated subfolders in the drop folder
        drop_folder = self.config["drop_folder"]
        date_pattern = re.compile(self.config["date_pattern"])

        try:
            subfolders = self.graph.list_subfolders(drop_folder)
        except Exception as e:
            self.logger.error(f"Failed to list subfolders in {drop_folder}: {e}")
            return

        dated_folders = [f for f in subfolders if date_pattern.match(f["name"])]
        self.logger.info(f"Found {len(dated_folders)} dated subfolders")

        # Step 2: Process each dated folder
        for folder in dated_folders:
            folder_name = folder["name"]
            folder_path = f"{drop_folder}/{folder_name}"

            folder_state = self.state.get(folder_name, {})
            status = folder_state.get("status", "unknown")

            if status == "processed":
                continue  # Already done

            if status == "pending":
                # Check if 30-minute wait is over
                first_seen = datetime.fromisoformat(folder_state["first_seen"])
                wait_until = first_seen + timedelta(minutes=self.config["wait_minutes"])
                now = datetime.now(timezone.utc)

                if now < wait_until:
                    remaining = (wait_until - now).total_seconds() / 60
                    self.logger.info(
                        f"Folder {folder_name}: waiting ({remaining:.1f} min remaining)"
                    )
                    continue
                else:
                    # Wait is over — process the folder
                    self._process_folder(folder_name, folder_path)

            else:
                # New folder — check for files and start the timer
                self._detect_new_folder(folder_name, folder_path)

        self.logger.info("Agent run completed")

    def _detect_new_folder(self, folder_name: str, folder_path: str):
        """Handle a newly detected dated subfolder."""
        try:
            files = self.graph.list_files(folder_path, extension=".csv")
        except Exception as e:
            self.logger.error(f"Failed to list files in {folder_path}: {e}")
            return

        if not files:
            self.logger.debug(f"Folder {folder_name}: no CSV files yet, ignoring")
            return

        # Files found — start the 30-minute timer
        now = datetime.now(timezone.utc).isoformat()
        self.state[folder_name] = {
            "status": "pending",
            "first_seen": now,
            "file_count_at_detection": len(files),
        }
        save_state(self.state_path, self.state)
        self.logger.info(
            f"New folder detected: {folder_name} with {len(files)} CSV files. "
            f"Waiting {self.config['wait_minutes']} minutes before processing."
        )

    def _process_folder(self, folder_name: str, folder_path: str):
        """Download all CSVs from a dated folder, consolidate, and upload."""
        self.logger.info(f"Processing folder: {folder_name}")

        # List all CSV files
        try:
            files = self.graph.list_files(folder_path, extension=".csv")
        except Exception as e:
            self.logger.error(f"Failed to list files in {folder_path}: {e}")
            return

        if not files:
            self.logger.warning(f"Folder {folder_name}: no CSV files found. Marking as processed.")
            self.state[folder_name]["status"] = "processed"
            self.state[folder_name]["processed_at"] = datetime.now(timezone.utc).isoformat()
            self.state[folder_name]["file_count"] = 0
            save_state(self.state_path, self.state)
            return

        self.logger.info(f"Downloading {len(files)} CSV files from {folder_name}")

        # Download all CSV files
        csv_contents: list[tuple[str, bytes]] = []
        for file_info in files:
            try:
                content = self.graph.download_file(file_info["id"])
                csv_contents.append((file_info["name"], content))
                self.logger.debug(f"Downloaded {file_info['name']} ({file_info['size']} bytes)")
            except Exception as e:
                self.logger.warning(f"Failed to download {file_info['name']}: {e}")

        if not csv_contents:
            self.logger.error(f"Folder {folder_name}: all file downloads failed")
            return

        # Consolidate CSVs
        try:
            consolidated = consolidate_csvs(
                csv_contents,
                add_source_metadata=self.config.get("add_source_metadata", False),
                logger=self.logger,
            )
        except ValueError as e:
            self.logger.error(f"Consolidation failed for {folder_name}: {e}")
            return

        # Upload consolidated CSV
        output_folder = self.config["output_folder"]
        output_filename = f"consolidated_{folder_name}.csv"

        try:
            self.graph.upload_file_with_auth(output_folder, output_filename, consolidated)
            self.logger.info(
                f"Successfully uploaded {output_filename} to {output_folder} "
                f"({len(consolidated)} bytes)"
            )
        except Exception as e:
            self.logger.error(f"Failed to upload {output_filename}: {e}")
            # Save locally as fallback
            fallback_path = os.path.join(tempfile.gettempdir(), output_filename)
            with open(fallback_path, "wb") as f:
                f.write(consolidated)
            self.logger.error(f"Saved fallback copy to {fallback_path}")
            return

        # Mark as processed
        self.state[folder_name] = {
            **self.state.get(folder_name, {}),
            "status": "processed",
            "processed_at": datetime.now(timezone.utc).isoformat(),
            "file_count": len(csv_contents),
            "output_file": output_filename,
            "output_size_bytes": len(consolidated),
        }
        save_state(self.state_path, self.state)

    def _handle_reprocess(self, date_str: str):
        """Force reprocessing of a specific dated folder."""
        self.logger.info(f"Reprocessing requested for {date_str}")

        # Reset state for this folder
        if date_str in self.state:
            self.state[date_str]["status"] = "ready"
            save_state(self.state_path, self.state)

        folder_path = f"{self.config['drop_folder']}/{date_str}"
        self._process_folder(date_str, folder_path)

def main():
    parser = argparse.ArgumentParser(
        description="SharePoint CSV Consolidation Agent",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog=__doc__,
    )
    parser.add_argument(
        "--config", "-c",
        default="config.json",
        help="Path to configuration JSON file (default: config.json)",
    )
    parser.add_argument(
        "--reprocess",
        metavar="DATE",
        help="Force reprocess a specific dated folder (e.g., 2026-03-18)",
    )
    args = parser.parse_args()

    # Load config
    config = load_config(args.config)

    # Validate required config
    required = ["tenant_id", "client_id", "client_secret", "site_url"]
    missing = [k for k in required if not config.get(k)]
    if missing:
        print(f"ERROR: Missing required configuration: {', '.join(missing)}", file=sys.stderr)
        print("Set via config.json or environment variables (SP_TENANT_ID, etc.)", file=sys.stderr)
        sys.exit(1)

    # Set up logging
    logger = setup_logging(config)

    # Acquire file lock to prevent concurrent runs
    lock = FileLock(config["lock_file"])
    if not lock.acquire():
        logger.info("Another instance is running (lock file exists). Exiting.")
        sys.exit(0)

    try:
        agent = SharePointCSVAgent(config, logger)
        agent.run(reprocess_date=args.reprocess)
    except Exception as e:
        logger.error(f"Unhandled exception: {e}", exc_info=True)
        sys.exit(1)
    finally:
        lock.release()


if __name__ == "__main__":
    main()
